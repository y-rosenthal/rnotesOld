drawSomething = function(size){
  
  if (!is.numeric(size) || trunc(size) != size || size < 0){
    stop("size must be a non-negative whole number")
  }
  
  for(row in 1:size){

    if( row == 1 || row == size ){
      for (x in 1:size){
        cat("-")
      }
      cat("\n")
      
    } else {
      for (col in 1:size) {
        if( row == 1 || row == size ){
          for (x in 1:size){
            cat("-")
          }
          cat("\n")
          
        } else {
          if (col == 1){
            cat("|")
          } else if (col == size){
            cat("|\n")
          } else if (row == col) {
            cat("x")
          } else {
            cat(".")
          }
        }
      }
    }

  }
}
